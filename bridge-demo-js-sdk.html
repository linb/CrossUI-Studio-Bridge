CrossUI Studio Snippet Bridge (JS SDK) Tester

This page demonstrates the professional JS SDK integration. The script automatically scans the page and adds "Open in Studio" buttons to matching code blocks.

Automatic Component Detection
import React from 'react';
import { Box, Typography } from '@mui/material';

export default function HelloAuto() {
    return (
        <Box sx={{ p: 4, bgcolor: 'primary.main', borderRadius: 2 }}>
            <Typography variant="h5" color="white">
                Automatically Detected Snippet
            </Typography>
        </Box>
    );
}
